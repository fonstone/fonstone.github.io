## 1. 内核链接文件

内核编译后生成的目标文件是 ELF 格式的 vmlinux，它是各源代码按照 `vmlinux.lds` 设定的规则链接后得到的 Object 文件，并不能直接在 ARM 平台上运行。通常内核映像以压缩格式存储（zImage 或 bzImage），因此内核启动阶段需要先自解压。

Linux 内核的链接文件位于 `arch/arm64/kernel/vmlinux.lds.S`，编译时会据此生成 `vmlinux.lds`，用于最终链接生成内核镜像 `vmlinux`。

在该文件的开头指定了输出的架构和内核入口地址：

```
OUTPUT_ARCH(aarch64)
ENTRY(_text)
```

`_text` 是代码段的起始地址，定义在 `SECTIONS` 部分：

```
. = KIMAGE_VADDR + TEXT_OFFSET;

.head.text : {
    _text = .;
    HEAD_TEXT
}
```

其中 `KIMAGE_VADDR` 和 `TEXT_OFFSET` 定义在 `arch/arm64/include/asm/memory.h` 中：

![KIMAGE_VADDR 和 TEXT_OFFSET 宏定义](/images/linux-kernel-boot/kimage-vaddr.png)

## 2. 内核启动第一阶段（汇编阶段）

### 2.1 内核启动入口点

将编译好的 `vmlinux` ELF 文件反汇编后，可以查看入口地址：

![vmlinux 入口地址反汇编结果](/images/linux-kernel-boot/vmlinux-entry.png)

Linux 内核的第一条指令是 `add x13, x18, #0x16`，对应符号在 `.head.text` 段。在 `include/linux/init.h` 中有如下定义：

```c
/* For assembly routines */
#define __HEAD      .section ".head.text","ax"
#define __INIT       .section ".init.text","ax"
```

`__HEAD` 宏代表 `.head.text` 段，在 `arch/arm64/kernel/head.S` 中使用：

```asm
__HEAD
_head:
    /*
     * The requirements are:
     *   MMU = off, D-cache = off, I-cache = on or off,
     *   x0 = physical address to the FDT blob.
     */
#ifdef CONFIG_EFI
    add x13, x18, #0x16       // "MZ" signature required by UEFI
    b   primary_entry
#else
    b   primary_entry           // branch to kernel start, magic
    .long   0                   // reserved
#endif
    .quad   0                   // Image load offset from start of RAM
    le64sym _kernel_size_le     // Effective size of kernel image
    le64sym _kernel_flags_le    // Informative flags
    .quad   0                   // reserved
    .quad   0                   // reserved
    .quad   0                   // reserved
    .ascii  ARM64_IMAGE_MAGIC   // Magic number
#ifdef CONFIG_EFI
    .long   pe_header - _head   // Offset to PE header
pe_header:
    __EFI_PE_HEADER
#else
    .long   0                   // reserved
#endif
```

内核启动前需要关闭 MMU 和 D-cache，x0 为 FDT blob 的物理地址。`add x13, x18, #0x16` 用于形成 UEFI 要求的 "MZ" 签名，然后执行 `b primary_entry` 跳转。

> D-cache 可能缓存了 bootloader 的数据，不清除可能导致内核访问错误数据。而 bootloader 的指令与内核指令无关，所以可以不关闭 I-cache。

### 2.2 primary_entry 函数

```asm
SYM_CODE_START(primary_entry)
    bl  preserve_boot_args
    bl  init_kernel_el              // w0=cpu_boot_mode
    adrp x23, __PHYS_OFFSET
    and x23, x23, MIN_KIMG_ALIGN - 1  // KASLR offset, defaults to 0
    bl  set_cpu_boot_mode_flag
    bl  __create_page_tables
    bl  __cpu_setup                  // initialise processor
    b   __primary_switch
SYM_CODE_END(primary_entry)
```

`primary_entry` 依次调用以下函数：

#### 2.2.1 preserve_boot_args

```asm
SYM_CODE_START_LOCAL(preserve_boot_args)
    mov x21, x0                      // x21=FDT，将 dtb 地址暂存到 x21
    adr_l x0, boot_args             // 将 boot_args 数组地址保存到 x0
    stp x21, x1, [x0]               // 保存 x21、x1 到 boot_args[0]、boot_args[1]
    stp x2, x3, [x0, #16]           // 保存 x2、x3 到 boot_args[2]、boot_args[3]
    dmb sy                           // 内存屏障
    mov x1, #0x20                    // 4 x 8 bytes = 32
    b   __inval_dcache_area          // 无效化数据缓存区域
SYM_CODE_END(preserve_boot_args)
```

`boot_args` 定义在 `arch/arm64/setup.c` 中，用于保存内核启动时传入的参数（x0-x3）。

#### 2.2.2 init_kernel_el

这段代码根据当前异常级别（EL2 或 EL1）对处理器进行初始化：

```asm
SYM_FUNC_START(init_kernel_el)
    mov_q x0, INIT_SCTLR_EL1_MMU_OFF
    msr  sctlr_el1, x0              // 配置 EL1 系统控制寄存器

    mrs  x0, CurrentEL               // 读取当前异常级别
    cmp  x0, #CurrentEL_EL2          // 是否为 EL2？
    b.eq init_el2                     // 是则跳转到 EL2 初始化

init_el1:
    isb
    mov_q x0, INIT_PSTATE_EL1
    msr  spsr_el1, x0                // 配置 EL1 的 PSTATE
    msr  elr_el1, lr                  // 配置 EL1 异常链接寄存器
    mov  w0, #BOOT_CPU_MODE_EL1
    eret

init_el2:
    mov_q x0, HCR_HOST_NVHE_FLAGS
    msr  hcr_el2, x0                 // 配置 EL2 Hypervisor 配置寄存器
    isb
    init_el2_state

    adr_l x0, __hyp_stub_vectors     // EL2 异常向量表地址
    msr  vbar_el2, x0                // 设置 EL2 向量基址寄存器
    isb

    msr  elr_el2, lr
    mov  w0, #BOOT_CPU_MODE_EL2
    eret
SYM_FUNC_END(init_kernel_el)
```

<MermaidDiagram chart={`flowchart TD
    A["init_kernel_el"] --> B["配置 sctlr_el1"]
    B --> C{"当前异常级别?"}
    C -->|"EL1"| D["配置 PSTATE_EL1<br/>配置 elr_el1<br/>eret 返回"]
    C -->|"EL2"| E["配置 HCR_EL2<br/>init_el2_state<br/>设置异常向量表 vbar_el2<br/>eret 返回"]
    D --> F["返回 BOOT_CPU_MODE_EL1"]
    E --> G["返回 BOOT_CPU_MODE_EL2"]
    style C fill:#fff9c4
`} />

主要步骤：

1. 配置 EL1 的系统控制寄存器 `sctlr_el1`
2. 读取当前异常级别
3. 若为 EL1：配置 PSTATE 和异常链接寄存器后 `eret` 返回
4. 若为 EL2：配置 HCR 寄存器、初始化 EL2 状态、设置异常向量表后 `eret` 返回

常量定义：

```c
// arch/arm64/include/asm/virt.h
#define BOOT_CPU_MODE_EL1   (0xe11)
#define BOOT_CPU_MODE_EL2   (0xe12)
```

#### 2.2.3 set_cpu_boot_mode_flag

```asm
SYM_FUNC_START_LOCAL(set_cpu_boot_mode_flag)
    adr_l x1, __boot_cpu_mode
    cmp  w0, #BOOT_CPU_MODE_EL2
    b.ne 1f
    add  x1, x1, #4              // EL2 则偏移 4 字节
1:  str  w0, [x1]                 // 存储启动模式
    dmb  sy
    dc   ivac, x1                 // 无效化缓存行
    ret
SYM_FUNC_END(set_cpu_boot_mode_flag)
```

`__boot_cpu_mode` 是一个 64 位全局变量，前 4 字节默认值为 `0xe11`（EL1），后 4 字节为 `0xe12`（EL2）。根据实际启动级别写入对应位置。

#### 2.2.4 __create_page_tables

```asm
SYM_FUNC_START_LOCAL(__create_page_tables)
    mov  x28, lr

    // 1. 无效化 init 页表缓存
    adrp x0, init_pg_dir
    adrp x1, init_pg_end
    sub  x1, x1, x0
    bl   __inval_dcache_area

    // 2. 清空 init 页表
    adrp x0, init_pg_dir
    adrp x1, init_pg_end
    sub  x1, x1, x0
1:  stp  xzr, xzr, [x0], #16
    stp  xzr, xzr, [x0], #16
    stp  xzr, xzr, [x0], #16
    stp  xzr, xzr, [x0], #16
    subs x1, x1, #64
    b.ne 1b

    mov  x7, SWAPPER_MM_MMUFLAGS

    // 3. 创建恒等映射（identity mapping）
    adrp x0, idmap_pg_dir
    adrp x3, __idmap_text_start

    // 配置 VA_BITS 和 T0SZ
    // ...

    // 4. 映射内核镜像
    adrp x0, init_pg_dir
    mov_q x5, KIMAGE_VADDR
    add  x5, x5, x23
    // ...

    // 5. 无效化页表缓存
    dmb  sy
    adrp x0, idmap_pg_dir
    // ...
    bl   __inval_dcache_area

    adrp x0, init_pg_dir
    // ...
    bl   __inval_dcache_area

    ret  x28
SYM_FUNC_END(__create_page_tables)
```

    B --> C["创建恒等映射<br/>identity mapping"]
    C --> D["映射内核镜像<br/>kernel mapping"]
    D --> E["再次无效化缓存"]

主要功能：缓存无效化 → 页表清空 → 虚拟地址配置 → 内核映射。创建两类映射：

- **恒等映射（identity mapping）**：使 MMU 开启前后物理地址和虚拟地址一致
- **内核线性映射**：映射内核镜像本身

#### 2.2.5 __cpu_setup

`__cpu_setup` 定义在 `arch/arm64/mm/proc.S` 中，负责在开启 MMU 前初始化处理器：

1. 无效化本地 TLB（`tlbi vmalle1`）
2. 启用浮点和 SIMD 单元（`cpacr_el1`）
3. 设置调试寄存器
4. 配置内存属性寄存器 `MAIR_EL1`
5. 配置翻译控制寄存器 `TCR_EL1`（包括地址空间大小、页大小等）
6. 设置物理地址大小和硬件访问标志
7. 准备 `SCTLR_EL1` 的初始值并返回

#### 2.2.6 __primary_switch

```asm
SYM_FUNC_START_LOCAL(__primary_switch)
#ifdef CONFIG_RANDOMIZE_BASE
    mov  x19, x0                  // 保存新 SCTLR_EL1
    mrs  x20, sctlr_el1           // 保存旧 SCTLR_EL1
#endif
    adrp x1, init_pg_dir
    bl   __enable_mmu
#ifdef CONFIG_RELOCATABLE
    // 内核地址重定位...
#endif
    ldr  x8, =__primary_switched
    adrp x0, __PHYS_OFFSET
    br   x8
SYM_FUNC_END(__primary_switch)
```

主要工作是启用 MMU，必要时进行内核地址重定位（KASLR），然后跳转到 `__primary_switched`。

### 2.3 __primary_switched

```asm
SYM_FUNC_START_LOCAL(__primary_switched)
    adrp x4, init_thread_union
    add  sp, x4, #THREAD_SIZE
    adr_l x5, init_task
    msr  sp_el0, x5               // 保存 thread_info

    adr_l x8, vectors              // 加载异常向量表
    msr  vbar_el1, x8
    isb

    stp  xzr, x30, [sp, #-16]!
    mov  x29, sp

    str_l x21, __fdt_pointer, x5   // 保存 FDT 指针

    ldr_l x4, kimage_vaddr
    sub  x4, x4, x0                // 计算内核虚拟地址偏移
    str_l x4, kimage_voffset, x5

    // 清空 BSS 段
    adr_l x0, __bss_start
    mov  x1, xzr
    adr_l x2, __bss_stop
    sub  x2, x2, x0
    bl   __pi_memset
    dsb  ishst

    // KASAN 早期初始化
#if defined(CONFIG_KASAN_GENERIC) || defined(CONFIG_KASAN_SW_TAGS)
    bl   kasan_early_init
#endif
    mov  x0, x21                    // 传递 FDT 地址
    bl   early_fdt_map               // 早期映射 FDT
    bl   init_feature_override       // 解析 CPU 特性覆盖

#ifdef CONFIG_RANDOMIZE_BASE
    // KASLR 处理...
#endif

    bl   switch_to_vhe              // 优先使用 VHE
    add  sp, sp, #16
    mov  x29, #0
    mov  x30, #0
    b    start_kernel               // 跳转到 C 语言入口
SYM_FUNC_END(__primary_switched)
```

## 3. 内核启动第二阶段（C 语言阶段）

第二阶段从 `start_kernel()` 函数开始，定义在 `init/main.c` 中。这是所有 Linux 平台进入系统内核初始化后的入口函数，主要完成各子系统的初始化。

### 3.1 start_kernel


整体启动流程可以概括为三个阶段：

1. **汇编阶段**（`_head` → `__primary_switched`）：MMU 关闭状态下完成硬件初始化、建立页表、启用 MMU，并跳转到 C 语言入口
2. **C 语言阶段**（`start_kernel`）：依次初始化各内核子系统（内存、调度、中断、时间、VFS、安全等）
3. **用户空间启动**（`rest_init` → `kernel_init`）：创建 init 进程和 kthreadd 内核线程，最终启动用户空间
